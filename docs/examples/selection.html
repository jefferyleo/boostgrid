const [grid] = attach("#grid");

// Read currently selected identifier values
const ids = grid.getSelectedRows();
// → [1, 4, 12]

// Programmatic select / deselect
grid.select([1, 4, 12]);
grid.deselect();           // all

// Remove selected rows from the data
grid.remove();              // uses current selection

// Listen for selection changes
grid.on("selected",   (rows) => console.log("+", rows));
grid.on("deselected", (rows) => console.log("-", rows));
import { useRef } from "react";
import { ReactBoostgrid, type ReactBoostgridHandle } from "react-boostgrid";

const ref = useRef<ReactBoostgridHandle>(null);

<ReactBoostgrid
  ref={ref}
  data={rows}
  columns={columns}
  options={{
    selection:     true,
    multiSelect:   true,
    rowSelect:     true,
    keepSelection: true,
  }}
  onSelected={(rows) => console.log("+", rows)}
  onDeselected={(rows) => console.log("-", rows)}
/>

// Imperative escape hatches via the ref handle:
ref.current?.getSelectedRows();   // [1, 4, 12]
ref.current?.grid?.select([1, 4]);
