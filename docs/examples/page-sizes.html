Custom page sizes

Pass an array to rowCount to control the page-size dropdown. Use -1 to mean "All". The first entry becomes the initial size.

ID Sender Subject Received 
<table id="grid" class="table">…</table>
attach("#grid", {
  rowCount: [5, 10, 25, -1],   // -1 = "All"
});

// Or programmatically pick a size — the dropdown updates too:
grid.setRowCount?.(25);
<ReactBoostgrid
  data={rows}
  columns={cols}
  options={{ rowCount: [5, 10, 25, -1] }}
/>
<!-- PageSizeGrid.vue -->
<script setup lang="ts">
import { VueBoostgrid } from "vue-boostgrid";
</script>

<template>
  <VueBoostgrid
    :data="rows"
    :columns="cols"
    :options="{ rowCount: [5, 10, 25, -1] }"
  />
</template>
