<script setup lang="ts">
import { ref } from "vue";
import { VueBoostgrid, type VueBoostgridHandle } from "vue-boostgrid";

const handle = ref<VueBoostgridHandle | null>(null);
const rows = ref(initial);

const add    = () => rows.value = [...rows.value, newRow];
const clear  = () => rows.value = [];
const search = () => handle.value?.search("alpha");
const reload = () => handle.value?.reload();
</script>

<template>
  <button @click="add">Append</button>
  <button @click="clear">Clear</button>
  <button @click="search">Search</button>
  <button @click="reload">Reload</button>

  <VueBoostgrid ref="handle" :data="rows" :columns="cols" />
</template>
