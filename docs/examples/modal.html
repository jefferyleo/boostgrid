Inside a Bootstrap 5 modal

Two gotchas to know up-front:

  1. If the modal is rendered inside a parent with transform, filter or contain set, the modal's fixed positioning gets trapped and the backdrop covers it. Move the element to document.body before showing.
  2. Attach the grid after the dialog is visible — listen for shown.bs.modal. Mounting against a display: none ancestor makes the toolbar measure 0 px.

Also pass navigation: 2 if you only want the bottom pager — by default Boostgrid renders top + bottom (navigation: 3).

const modal = document.getElementById("ex-modal");

// 1. Portal the modal to <body> if a parent breaks fixed positioning.
if (modal.parentElement !== document.body) {
  document.body.appendChild(modal);
}

let grid;

// 2. Defer attach() until the dialog is actually visible — otherwise the grid
//    measures 0px and the toolbar/pagination render badly.
modal.addEventListener("shown.bs.modal", () => {
  if (!grid) {
    [grid] = attach("#grid", {
      selection:   true,
      multiSelect: true,
      navigation:  2,        // bottom toolbar only (1=top, 2=bottom, 3=both)
    });
  } else {
    grid.reload();           // re-render in case viewport changed
  }
});

<script setup lang="ts">
import { ref, watch } from "vue";
import { VueBoostgrid, type VueBoostgridHandle } from "vue-boostgrid";

const open = ref(false);
const handle = ref<VueBoostgridHandle | null>(null);

// Reload after the modal animates in
watch(open, (isOpen) => {
  if (isOpen) setTimeout(() => handle.value?.reload(), 200);
});
</script>

<template>
  <div :class="['modal fade', open ? 'show d-block' : '']">
    <div class="modal-dialog modal-xl">
      <div class="modal-content">
        <div class="modal-body">
          <VueBoostgrid ref="handle" :data="rows" :columns="cols" />
        </div>
      </div>
    </div>
  </div>
</template>
