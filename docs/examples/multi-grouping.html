Multi-level grouping

Group rows by one column, two, three — same option, same dataset. This sales-pipeline view nests Region → Stage → Deal; each level renders its own header with a row count, and groupAggregators use the colId@N suffix so the regional total reads at the outer tier and the stage subtotal sums inside it. The Subtotals on top toggle flips footer placement via the new groupSubtotalsOnTop option.

import { attach } from "boostgrid";

const [grid] = attach("#grid", {
  groupBy: ["region", "stage"],
  groupExpanded: "all",
  groupSubtotalsOnTop: false, // ← flip for headline-first layout

  groupAggregators: {
    // Region tier (depth 0): pipeline count + grand total
    "deal@0":   (_col, ctx) =>
      `<em class="text-muted">${ctx.rows.length} deals</em>`,
    "amount@0": (_col, ctx) => {
      const total = ctx.rows.reduce((s, r) => s + r.amount, 0);
      return `<strong>${money(total)}</strong>`;
    },

    // Stage tier (depth 1): sum + average within the stage
    "amount@1": (_col, ctx) => {
      const sum = ctx.rows.reduce((s, r) => s + r.amount, 0);
      const avg = sum / ctx.rows.length;
      return `<span class="text-muted small">avg ${money(avg)}</span>
                <strong>${money(sum)}</strong>`;
    },
  },
});

// Switch grouping at runtime — same data, different shape
grid.options.groupBy = ["region", "stage", "owner"];
grid.rerenderBody();
// Aggregator key resolution at depth N:
//   1. "colId@N"  — wins (explicit depth match)
//   2. "colId"    — fallback (runs at every depth)

groupAggregators: {
  // Runs at depth 0 only — region totals
  "amount@0": (col, ctx) => `<strong>$${sum(ctx.rows)}</strong>`,

  // Runs at depth 1 only — per-stage sums
  "amount@1": (col, ctx) => `$${sum(ctx.rows)}`,

  // Runs at every depth where there's no @N override —
  //  here, depth 2+ if you nest a third level
  "amount":    (col, ctx) => `<em>$${sum(ctx.rows)}</em>`,
}

// `ctx.rows` always contains every leaf row in this group AND
// its descendants — so a depth-0 aggregator sees the full
// regional slice, not just the rows directly below the header.

<script setup lang="ts">
import { VueBoostgrid } from "vue-boostgrid";

const cols = [
  { id: "id",     text: "ID",     identifier: true, type: "numeric" },
  { id: "region", text: "Region" },
  { id: "stage",  text: "Stage" },
  { id: "deal",   text: "Deal" },
  { id: "amount", text: "Amount", type: "numeric" },
];

const opts = {
  groupBy: ["region", "stage"],
  groupSubtotalsOnTop: true,
  groupAggregators: {
    "amount@0": (_c, ctx) => `<strong>$${sum(ctx.rows)}</strong>`,
    "amount@1": (_c, ctx) => `$${sum(ctx.rows)}`,
  },
};
</script>

<template>
  <VueBoostgrid :data="deals" :columns="cols" :options="opts" />
</template>
