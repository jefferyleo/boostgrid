React wrapper

The react-boostgrid package mounts a Boostgrid instance for you. React renders an empty host <div>; the table is built imperatively so the reconciler never fights boostgrid's DOM mutations. 

import { useRef, useState } from "react";
import { ReactBoostgrid, type ReactBoostgridHandle } from "react-boostgrid";
import "boostgrid/style.css";

export default function Inbox() {
  const [rows, setRows] = useState(initial);
  const ref = useRef<ReactBoostgridHandle>(null);

  return (
    <ReactBoostgrid
      ref={ref}
      data={rows}
      columns={[
        { id: "id",      text: "ID",      identifier: true, type: "numeric", align: "right" },
        { id: "sender",  text: "Sender",  order: "asc" },
        { id: "subject", text: "Subject" },
      ]}
      options={{ rowCount: 10, selection: true, multiSelect: true }}
      onSelected={(rows) => console.log("selected", rows)}
    />
  );
}
