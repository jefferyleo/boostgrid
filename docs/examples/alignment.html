Header & row align

data-align sets the cell alignment, data-header-align the header alignment. Each defaults to left; values are left · center · right.

ID Sender (left) Subject (centred) Received (header L · cell R) Status 
<!-- Header and cell alignment are independent -->
<th data-column-id="id"
    data-align="right"
    data-header-align="right">ID</th>

<th data-column-id="received"
    data-align="right"
    data-header-align="left">Received</th>

<th data-column-id="subject"
    data-align="center"
    data-header-align="center">Subject</th>

<!-- Or via cssClass / headerCssClass for full BS5 utilities -->
<th data-column-id="status"
    data-css-class="text-uppercase"
    data-header-css-class="fw-bold">Status</th>
// All column alignment is data-driven — no JS configuration required.
attach("#grid");

// Numeric columns auto-right by convention; you can still override:
attach("#grid", {
  converters: {
    currency: {
      from: (v) => Number(v),
      to:   (v) => v.toLocaleString("en", { style: "currency", currency: "USD" }),
    },
  },
});
// Alignment is configured directly on each column object.
<ReactBoostgrid
  data={rows}
  columns={[
    { id: "id",       text: "ID",       identifier: true, type: "numeric",
      align: "right", headerAlign: "right" },
    { id: "sender",   text: "Sender",
      align: "left",  headerAlign: "left" },
    { id: "subject",  text: "Subject",
      align: "center", headerAlign: "center" },
    { id: "received", text: "Received",
      align: "right", headerAlign: "left" },
    { id: "status",   text: "Status",
      align: "center", headerAlign: "center",
      cssClass: "text-uppercase", headerCssClass: "fw-bold" },
  ]}
