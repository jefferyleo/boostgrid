Inline row edit

Each cell is rendered through a formatter that returns an <input>. A delegated change handler on the table element captures edits and updates the row data — the grid's internal state is the source of truth.

const [grid] = attach("#grid", {
  formatters: {
    editText(col, row) {
      const v = String(row[col.id] ?? "").replace(/"/g, "&quot;");
      return `<input class="form-control form-control-sm"
          data-edit-id="${row.id}"
          data-edit-field="${col.id}"
          value="${v}">`;
    },
  },
});

// One delegated listener handles every editable cell.
grid.element.addEventListener("change", (e) => {
  const inp = e.target.closest("input[data-edit-field]");
  if (!inp) return;

  const id    = Number(inp.dataset.editId);
  const field = inp.dataset.editField;
  const row   = rows.find((r) => r.id === id);
  if (row) row[field] = inp.value;

  savePatch(id, { [field]: inp.value });
});

useEffect(() => {
  const el = ref.current?.grid?.element;
  if (!el) return;
  const on = (e) => {
    const inp = e.target.closest("input[data-edit-field]");
    if (!inp) return;
    const id    = Number(inp.dataset.editId);
    const field = inp.dataset.editField;
    setRows((rs) => rs.map((r) => r.id === id ? { ...r, [field]: inp.value } : r));
    savePatch(id, { [field]: inp.value });
  };
  el.addEventListener("change", on);
  return () => el.removeEventListener("change", on);
}, []);

<script setup lang="ts">
import { ref, onMounted, onBeforeUnmount } from "vue";
import { VueBoostgrid, type VueBoostgridHandle } from "vue-boostgrid";

const handle = ref<VueBoostgridHandle | null>(null);
const rows = ref(initial);

const editText = (col, row) => {
  const v = String(row[col.id] ?? "").replace(/"/g, "&quot;");
  return `<input class="form-control form-control-sm"
      data-edit-id="${row.id}"
      data-edit-field="${col.id}"
      value="${v}">`;
};

let el: HTMLElement | null = null;
const on = (e) => {
  const inp = e.target.closest("input[data-edit-field]");
  if (!inp) return;
  const id    = Number(inp.dataset.editId);
  const field = inp.dataset.editField;
  rows.value = rows.value.map((r) => r.id === id ? { ...r, [field]: inp.value } : r);
  savePatch(id, { [field]: inp.value });
};
onMounted(() => {
  el = handle.value?.grid?.element ?? null;
  el?.addEventListener("change", on);
});
onBeforeUnmount(() => el?.removeEventListener("change", on));

const cols = [
  { id: "id",      text: "ID",      identifier: true, type: "numeric", align: "right" },
  { id: "sender",  text: "Sender",  formatter: editText },
  { id: "subject", text: "Subject", formatter: editText },
];
</script>

<template>
  <VueBoostgrid ref="handle" :data="rows" :columns="cols" />
</template>
