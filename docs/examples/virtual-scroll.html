Virtual scroll · 10,000 rows

Set virtualScroll: true with a fixed rowHeight and Boostgrid renders only the rows in the viewport plus an overscan on each side. Pad <tr>s above and below the slice preserve the scrollbar shape. Since v2.5.0, the visible <tr> elements are recycled as a pool on every scroll tick instead of being rebuilt, and scrollToRow(index) exposes jump-scroll on the public API.

10,000 rows in the dataset · row height 32px · overscan 5 ·
ID Sender Subject Received 
const [grid] = attach("#grid", {
  virtualScroll: true,
  rowHeight:     32,    // must match the actual row height
  overscan:      5,     // rows rendered above + below the viewport
});

// rowsPerPage is forced to -1 internally — virtual scroll is the slicing layer.

// Jump to an arbitrary row programmatically (2.5.0+):
grid.scrollToRow(4999);   // clamped to [0, total - 1]; returns `this`.

.boostgrid--virtual .boostgrid-table tbody {

}
import { useRef } from "react";
import { ReactBoostgrid, type ReactBoostgridHandle } from "react-boostgrid";

const ref = useRef<ReactBoostgridHandle>(null);

<ReactBoostgrid
  ref={ref}
  data={tenThousandRows}
  columns={cols}
  options={{
    virtualScroll: true,
    rowHeight:     32,
    overscan:      5,
  }}
/>

// Jump scroll via the ref handle (2.5.0+):

<script setup lang="ts">
import { ref } from "vue";
import { VueBoostgrid, type VueBoostgridHandle } from "vue-boostgrid";

const props = defineProps<{ tenThousandRows: any[] }>();
const handle = ref<VueBoostgridHandle | null>(null);
const cols = [
  { id: "id",       text: "ID",       identifier: true, type: "numeric" },
  { id: "sender",   text: "Sender" },
  { id: "subject",  text: "Subject" },
  { id: "received", text: "Received" },
];

// Jump scroll via the template ref handle (2.5.0+):
handle.value?.scrollToRow(4999);
</script>

<template>
  <VueBoostgrid
    ref="handle"
    :data="props.tenThousandRows"
    :columns="cols"
