Responsive

Wrap the table in .table-responsive for horizontal scroll on narrow screens. Give the table a min-width (or sum-of-column widths) so the overflow actually triggers — try resizing your window and watch the bottom scrollbar appear under ~1100 px.

<!-- Wrap with .table-responsive for x-scroll on narrow screens.
     The min-width is what makes the parent overflow trigger. -->
<div class="table-responsive">
  <table id="grid" class="table table-bordered"
         style="min-width: 1100px">
    <thead>
      <tr>
        <th data-column-id="id" data-identifier="true" data-width="80">ID</th>
        <th data-column-id="sender"  data-width="240">Sender</th>
        <th data-column-id="subject" data-width="380">Subject</th>
        <th data-column-id="received" data-width="180">Received</th>
        <th data-column-id="status"  data-width="120">Status</th>
      </tr>
    </thead>
    <tbody>…</tbody>
  </table>
</div>
