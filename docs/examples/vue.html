Vue 3 wrapper

vue-boostgrid mirrors the React wrapper's host-div pattern for Vue 3. Vue owns the `<div>` host; the `<table>` is built imperatively from the columns prop on mount, so Vue's reactivity layer never collides with the grid's DOM mutations.

Heads-up: the live demo below mounts a real Vue 3 app scoped to the host div. Vue is loaded lazily via esm.sh through the page's importmap, so the rest of the docs site pays no Vue cost. 
<!-- App.vue -->
<script setup lang="ts">
import { ref } from "vue";
import { VueBoostgrid } from "vue-boostgrid";
import "boostgrid/style.css";

const data = ref([
  { id: 1, sender: "alpha@x.com", subject: "Welcome" },
  { id: 2, sender: "beta@x.com",  subject: "Hello" },
]);
const cols = [
  { id: "id",      text: "ID",      identifier: true, type: "numeric" },
  { id: "sender",  text: "Sender",  order: "asc" },
  { id: "subject", text: "Subject" },
];
</script>

<template>
  <VueBoostgrid
    :data="data"
    :columns="cols"
    :options="{ stateSave: true, selection: true }"
    @selected="(rs) => console.log(rs)"
  />
</template>

// In your bundler (Vite, etc.):
import "boostgrid/style.css";
import { VueBoostgrid } from "vue-boostgrid";
import { ref } from "vue";
import { VueBoostgrid, type VueBoostgridHandle } from "vue-boostgrid";

const handle = ref<VueBoostgridHandle | null>(null);

// Same imperative API as the React wrapper:
handle.value?.search("alpha");
handle.value?.sort({ received: "desc" });
handle.value?.reload();
handle.value?.getSelectedRows();
handle.value?.grid;   // raw Boostgrid instance escape hatch
