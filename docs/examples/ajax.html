Server-side data (Ajax)

Set ajax: true and a url; Boostgrid will POST a JSON request body describing the current sort / page / filter state and expects a JSON envelope back. Customise either side via requestHandler / responseHandler.

Note: this docs site is static, so the live ajax demo is mocked — the contracts below are what your server should implement. 
<!-- The columns describe the *fields* on each returned row -->
<table id="grid" class="table table-hover">
  <thead>
    <tr>
      <th data-column-id="id" data-identifier="true" data-type="numeric">ID</th>
      <th data-column-id="sender" data-order="asc">Sender</th>
      <th data-column-id="subject">Subject</th>
    </tr>
  </thead>
  <tbody></tbody>
</table>
attach("#grid", {
  ajax: true,
  url:  "/api/messages",
  ajaxSettings: { method: "POST" },

  // Optionally enrich the request body
  requestHandler: (req) => ({ ...req, locale: "en" }),

  // Reshape the server response into the expected envelope
  responseHandler: (res) => ({
    current:  res.page,
    rowCount: res.size,
    total:    res.totalCount,
    rows:     res.items,
  }),
});
// In React the wrapper still drives the same ajax pipeline.
// Pass `data={[]}` to mount on an empty initial set; boostgrid will then
// fetch on first render via its built-in ajax flow.
<ReactBoostgrid
  data={[]}
  columns={[
    { id: "id",      text: "ID",      identifier: true, type: "numeric" },
    { id: "sender",  text: "Sender",  order: "asc" },
    { id: "subject", text: "Subject" },
  ]}
  options={{
    ajax: true,
    url:  "/api/messages",
    ajaxSettings: { method: "POST" },
    requestHandler:  (req) => ({ ...req, locale: "en" }),
    responseHandler: (res) => ({
      current:  res.page,
      rowCount: res.size,
      total:    res.totalCount,
      rows:     res.items,
    }),
  }}
/>
// POST /api/messages
// Content-Type: application/json
{
  "current":      2,
  "rowCount":     10,
  "sort":         { "sender": "asc", "received": "desc" },
  "searchPhrase": "alpha"
}
// 200 OK · application/json
{
  "current":  2,
  "rowCount": 10,
  "total":    137,
  "rows": [
    { "id": 11, "sender": "alpha@example.com", "subject": "Welcome",    "received": "2026-04-12" },
    { "id": 12, "sender": "alpha@example.com", "subject": "Newsletter", "received": "2026-04-09" }
  ]
}
