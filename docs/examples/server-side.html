import { attach } from "boostgrid";

attach("#grid", {
  ajax: true,
  url: "/api/rows",

  // Either or both — Boostgrid surfaces the active state in the
  // request body so your endpoint can return the matching slice.
  groupBy: ["status"],
  treeMode: false,

  // Skeleton placeholder rows render automatically while the
  // fetch is in flight.
  loadingSkeleton: true,
});

  
// AjaxRequest — fields are present only when their feature is on,
// so existing servers keep seeing the same shape.
{
  "current": 1,
  "rowCount": 25,
  "sort": { "received": "desc" },
  "searchPhrase": "",

  // Only when groupBy is set:
  "groupBy": ["status", "region"],
  "collapsedGroups": ["active//us"],

  // Only when treeMode is on:
  "treeMode": true,
  "expandedTreeNodes": [1, 4, 42]
}

  
// Express adapter sketch — adapt to your stack.
app.post("/api/rows", async (req, res) => {
  const { current, rowCount, sort, groupBy, treeMode, expandedTreeNodes } = req.body;

  let rows;
  if (treeMode) {
    // Return roots PLUS direct children of every expanded node.
    // Boostgrid re-builds the adjacency tree client-side.
    rows = await db.query(
      "SELECT * FROM nodes WHERE parent_id IS NULL OR parent_id = ANY($1)",
      [expandedTreeNodes ?? []],
    );
  } else if (groupBy?.length) {
    // Sort by groupBy fields first so the client's adjacency-based
    // bucketing produces correct headers in one pass.
    const orderBy = [...groupBy, ...Object.keys(sort)].join(", ");
    rows = await db.query(`SELECT * FROM deals ORDER BY ${orderBy}`);
  } else {
    const offset = (current - 1) * rowCount;
    rows = await db.query("SELECT * FROM deals LIMIT $1 OFFSET $2", [rowCount, offset]);
  }

  res.json({ current, rowCount, rows, total: rows.length });
});
