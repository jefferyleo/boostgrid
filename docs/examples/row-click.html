Row-click select (sticky)

With rowSelect: true the entire row body becomes the click target — not just the checkbox. Combined with keepSelection: true, your selection survives sorting, filtering, and pagination.

ID Sender Subject Received
Selected ids: [] 
const [grid] = attach("#grid", {
  selection:     true,
  multiSelect:   true,
  rowSelect:     true,    // click anywhere on a row
  keepSelection: true,    // survive paging / filter / sort
});

grid.on("selected", () => render(grid.getSelectedRows()));
grid.on("deselected", () => render(grid.getSelectedRows()));

<script setup lang="ts">
import { ref } from "vue";
import { VueBoostgrid, type VueBoostgridHandle } from "vue-boostgrid";

const handle = ref<VueBoostgridHandle | null>(null);
const ids = ref<number[]>([]);
</script>

<template>
  <VueBoostgrid
    ref="handle"
    :data="rows"
    :columns="cols"
    :options="{ selection: true, multiSelect: true, rowSelect: true, keepSelection: true }"
    @selected="(rs) => ids = rs.map((r) => r.id)"
    @deselected="() => ids = handle?.getSelectedRows() ?? []"
  />
</template>
