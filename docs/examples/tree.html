Tree data

Set treeMode: true and give each row a parentId field. Boostgrid reconstructs the adjacency-list tree, renders rows depth-first with an indentation per level, and adds an expand/collapse caret on the column you mark as treeColumn. Search reveals matched leaves and their ancestor chain so the path to a hit is always visible. State persistence remembers which branches you've collapsed.

import { attach } from "boostgrid";

// Each row has an id (from the identifier column) and a parentId.
// Roots use parentId === null.
const rows = [
  { id: 1, name: "src",        parentId: null },
  { id: 2, name: "core.ts",    parentId: 1 },
  { id: 3, name: "render",     parentId: 1 },
  { id: 4, name: "body.ts",    parentId: 3 },
  { id: 5, name: "tree.ts",    parentId: 3 },
];

const [grid] = attach("#grid", {
  treeMode: true,
  treeParentField: "parentId",   // default
  treeIndentPx: 22,             // default 24
  treeExpanded: "all",         // or "none" / Record<id, bool>
});
<!-- Mark the column that should carry the caret -->
<table id="grid" class="table"
       data-tree-mode="true">
  <thead>
    <tr>
      <th data-column-id="id" data-identifier="true" data-type="numeric">ID</th>
      <th data-column-id="name" data-tree-column="true">Name</th>
      <th data-column-id="parentId" data-type="numeric"
          data-visible="false">Parent</th>
    </tr>
  </thead>
  <tbody>…</tbody>
</table>
// Imperative API for tree mode
grid.toggleTreeNode(3);            // flip the "render" subtree
grid.isTreeExpanded(3);            // → boolean
grid.expandAllTree();
grid.collapseAllTree();

// Search across all visible columns; ancestors of matches stay visible
grid.search("body");
// → only "body.ts" matches, but "src" and "render" stay visible too
