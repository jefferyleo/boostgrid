Commands column

A formatter renders an Edit / Delete button group per row. The grid's root element receives delegated clicks on [data-bg-action] — the same mechanism the library uses internally for sort, paging and selection, so you don't need a per-row handler.

ID Sender Subject Actions

data-bg-action events:

<th data-column-id="commands"
    data-formatter="commands"
    data-sortable="false"
    data-searchable="false"
    data-align="center"
    data-width="120">Actions</th>
const [grid] = attach("#grid", {
  formatters: {
    commands(_col, row) {
      return `
        <div class="btn-group btn-group-sm" role="group">
          <button class="btn btn-outline-primary"
                  data-bg-action="cmd-edit" data-bg-value="${row.id}">
            <i class="bi bi-pencil"></i> Edit
          </button>
          <button class="btn btn-outline-danger"
                  data-bg-action="cmd-delete" data-bg-value="${row.id}">
            <i class="bi bi-trash"></i> Delete
          </button>
        </div>`;
    },
  },
});

// Single delegated listener — survives re-render, no per-row binding needed.
grid.element.addEventListener("click", (e) => {
  const btn = e.target.closest("[data-bg-action]");
  if (!btn) return;
  const id = Number(btn.dataset.bgValue);

  if (btn.dataset.bgAction === "cmd-edit")   openEditDialog(id);
  if (btn.dataset.bgAction === "cmd-delete") grid.remove([id]);
});

const commands = (_col, row) => `
  <div class="btn-group btn-group-sm">
    <button class="btn btn-outline-primary"
            data-bg-action="cmd-edit" data-bg-value="${row.id}">
      <i class="bi bi-pencil"></i>
    </button>
    <button class="btn btn-outline-danger"
            data-bg-action="cmd-delete" data-bg-value="${row.id}">
      <i class="bi bi-trash"></i>
    </button>
  </div>`;

useEffect(() => {
  const el = ref.current?.grid?.element;
  if (!el) return;
  const handler = (e) => {
    const btn = e.target.closest("[data-bg-action]");
    if (!btn) return;
    const id = Number(btn.dataset.bgValue);
    if (btn.dataset.bgAction === "cmd-edit")   openEditDialog(id);
    if (btn.dataset.bgAction === "cmd-delete") setRows((rs) => rs.filter((r) => r.id !== id));
  };
  el.addEventListener("click", handler);
  return () => el.removeEventListener("click", handler);
}, []);

  ref={ref}
  data={rows}

<script setup lang="ts">
import { ref, onMounted, onBeforeUnmount } from "vue";
import { VueBoostgrid, type VueBoostgridHandle } from "vue-boostgrid";

const handle = ref<VueBoostgridHandle | null>(null);
const rows = ref(initial);

const commands = (_col, row) => `
  <div class="btn-group btn-group-sm">
    <button class="btn btn-outline-primary"
            data-bg-action="cmd-edit" data-bg-value="${row.id}">
      <i class="bi bi-pencil"></i>
    </button>
    <button class="btn btn-outline-danger"
            data-bg-action="cmd-delete" data-bg-value="${row.id}">
      <i class="bi bi-trash"></i>
    </button>
  </div>`;

// Single delegated listener — survives re-render.
let el: HTMLElement | null = null;
const handler = (e) => {
  const btn = e.target.closest("[data-bg-action]");
  if (!btn) return;
  const id = Number(btn.dataset.bgValue);
  if (btn.dataset.bgAction === "cmd-edit")   openEditDialog(id);
  if (btn.dataset.bgAction === "cmd-delete") rows.value = rows.value.filter((r) => r.id !== id);
};
onMounted(() => {
  el = handle.value?.grid?.element ?? null;
  el?.addEventListener("click", handler);
});
onBeforeUnmount(() => el?.removeEventListener("click", handler));

const cols = [
  /* … other columns … */
  { id: "commands", text: "Actions", formatter: commands,
    sortable: false, searchable: false, align: "center", width: "120" },
];
</script>

<template>
  <VueBoostgrid ref="handle" :data="rows" :columns="cols" />
</template>
